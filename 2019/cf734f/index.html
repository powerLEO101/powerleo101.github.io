Codeforces734F Anton and School

Written by Leo101 with ♥ on May 2, 2019 in cjk贪心与构造 Codeforces

题面

Codeforces

Luogu

题解

总而言之这题是真的神仙

首先看到这两个式子

b[i]=(a[i] and a[1])+(a[i] and a[2])+...+(a[i] and a[n])
c[i]=(a[i] or a[1])+(a[i] or a[2])+...+(a[i] or a[n])

因为

(a and b) + (a or b) = a + b

所以

$$b[i] + c[i] = (\sum_{j = 1}^{n}{a[j]})+n*a[i]$$

所以

$$ \sum_{i=1}^{n}=2*n*\sum_{i=1}^{n}{a[i]} $$

所以

$$ \frac{\sum_{i = 1}^{n}{(b[i] +c[i])}}{2*n}{=\sum_{i = 1}^{n}{a[i]}} $$

然后我们就可以得到所有 a[i] 的和了

然后再根据

$$ b[i] + c[i] = (\sum_{j = 1}^{n}{a[j]})+n *a[i] $$

得到每一个 $a[i]$ 的值

接下来就是验证，常规的$O(n^2)$肯定不行，直接用位运算优化一下就可以了

对于与操作

a[i] 的某一位如果是 0 这一位得到的就都是 0 ，如果是 1 就得到所有这一位上的 1

对于或操作

a[i] 某一位如果是1这一位就有 n 个 1 ，是 0 就加上其他数的 1 ，实际上就是这一位上所以的 1

#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#define _FILE(s) freopen(#s".in", "r", stdin); freopen(#s".out", "w", stdout)
#define gi get_int()
typedef unsigned long long LL;
#define int LL
const int MAXN = 2e6;
int get_int()
{
	int x = 0, y = 1; char ch = getchar();
	while (!isdigit(ch) && ch != '-') ch = getchar();
	if (ch == '-') y = -1, ch = getchar();
	while (isdigit(ch)) x = x * 10 + ch - '0', ch = getchar();
	return x * y;
}

int numA[MAXN], numB[MAXN], numC[MAXN], count[64];

bool check(int n)
{
	// get sum
	for (int i = 0; i < n; i++)
		for (int j = 0; j < 64; j++) 
			count[j] += (numA[i] >> j) & 1;

	// operation &
	for (int i = 0; i < n; i++) {
		int temp = 0;
		for (int j = 0; j < 64; j++) {
			if (((numA[i] >> j) & 1) == 0) continue;
			temp += count[j] << j;
		}
		if (temp != numB[i]) return false;
	}

	// operation |
	for (int i = 0; i < n; i++) {
		int temp = 0;
		for (int j = 0; j < 64; j++) {
			if (((numA[i] >> j) & 1) == 1) {
				temp += n << j;
			} else {
				temp += count[j] << j;
			}
		}
		if (temp != numC[i]) return false;
	}

	return true;
}

signed main()
{
	_FILE(code);

	int n = gi;
	for (int i = 0; i < n; i++) numB[i] = gi;
	for (int i = 0; i < n; i++) numC[i] = gi;

	int sum = 0;
	for (int i = 0; i < n; i++) 
		sum += numB[i] + numC[i];
	sum /= 2 * n;

	for (int i = 0; i < n; i++) 
		numA[i] = ((numB[i] + numC[i]) - sum) / n;

	if (check(n) == false) {
		printf("-1");
		return 0;
	}

	for (int i = 0; i < n; i++) std :: cout << numA[i] << " ";

	return 0;
}
